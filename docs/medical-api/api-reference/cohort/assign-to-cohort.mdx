```javascript Metriport SDK
import { MetriportMedicalApi } from "@metriport/api-sdk";

const metriport = new MetriportMedicalApi("YOUR_API_KEY");

// Assign specific patients to a cohort
const cohortDetails =
  await metriport.assignPatientsToCohort("cohort-id-123", {
    patientIds: [
      "patient-id-1",
      "patient-id-2",
      "patient-id-3",
    ],
  });

// Or assign all patients to a cohort
const allPatientsCohort =
  await metriport.assignPatientsToCohort("cohort-id-123", {
    all: true,
  });
```
