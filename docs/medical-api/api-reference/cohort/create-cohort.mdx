```javascript Metriport SDK
import { MetriportMedicalApi } from "@metriport/api-sdk";

const metriport = new MetriportMedicalApi("YOUR_API_KEY");

const cohort = await metriport.createCohort({
  name: "High Risk",
  color: "red",
  description: "Patients that need frequent and robust monitoring.",
  settings: {
    monitoring: {
      hie: {
        enabled: true,
        frequency: "weekly"
      },
      adt: true,
      pharmacy: {
        notifications: true,
        schedule: {
          enabled: false,
          frequency: "monthly"
        }
      },
      laboratory: {
        notifications: true,
        schedule: {
          enabled: false,
          frequency: "monthly"
        }
      },
    },
  }
});
```
