const metriportClient = new MetriportMedicalApi("YOUR_API_KEY");

  type: OrgType.postAcuteCare,
  name: "Metriport Inc.",

    addressLine2: "#4818",
    city: "San Francisco",

    zip: "94114",
    country: "USA",
  }
});
