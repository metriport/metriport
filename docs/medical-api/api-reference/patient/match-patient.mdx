      },
    ],
    address: [
      {
        zip: "12345",

        country: "USA",
        addressLine1: "Guadalajara Street",
      },
    ],
    contact: [
      {
        phone: "1234567899",
        email: "jose@domain.com",
      },
    ],
  },
);
