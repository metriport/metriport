<ResponseField name="id" type="string" required>
  The unique identifier for the cohort (UUID format, e.g., `00000000-0000-0000-0000-000000000000`).
</ResponseField>

<ResponseField name="eTag" type="string" required>
  The entity tag for optimistic concurrency control.
</ResponseField>

<ResponseField name="name" type="string" required>
  The unique name of the cohort within your organization.
</ResponseField>

<ResponseField name="description" type="string" required>
  The description of the cohort.
</ResponseField>

<ResponseField name="color" type="string" required>
  The color associated with the cohort.
</ResponseField>

<ResponseField name="settings" type="object" required>
  Configuration settings for the cohort.
  <Expandable title="Settings properties">
    <ResponseField name="hie" type="object" optional>
      HIE-based document monitoring - subscribes patients to HIE-based document monitoring (non-ADT) / medical record refresh based on the cadence.
      <Expandable title="hie object properties" defaultOpen={false}>
        <ResponseField name="enabled" type="boolean" optional>
          Enable or disable scheduled data pulls for HIE monitoring. When `false`, no scheduled pulls will occur regardless of the frequency setting.
        </ResponseField>
        
        <ResponseField name="frequency" type="string" optional>
          How often scheduled data pulls should occur, can be one of `weekly` or `biweekly` or `monthly`.
        </ResponseField>
        
        <Info>
          See [When do scheduled pulls run?](/medical-api/handling-data/cohorts#when-do-scheduled-pulls-run%3F) for timing details.
        </Info>
      </Expandable>
      <Info>
        **HIE monitoring is coming soon!** While this feature is not yet live, you can configure your HIE monitoring settings now. These settings will automatically take effect when the feature launches.
      </Info>
    </ResponseField>
    <ResponseField name="adt" type="boolean" optional>
      ADT (Admission, Discharge, Transfer) monitoring - subscribes patients to ADT notifications based on their [primary address state](/medical-api/api-reference/patient/create-patient#param-is-primary).
    </ResponseField>
    <ResponseField name="pharmacy" type="object" optional>
      Pharmacy / medication history monitoring - enables retail fill data monitoring where available.  
      <Expandable title="pharmacy object properties">
        <ResponseField name="notifications" type="boolean" optional>
          Enable/disable pharmacy monitoring notifications.
        </ResponseField>
        <ResponseField name="schedule" type="object" optional>
          Enable pharmacy data pulls at scheduled dates.
          <Expandable title="schedule object properties" defaultOpen={false}>
            <ResponseField name="enabled" type="boolean" optional>
              Enable or disable scheduled data pulls for pharmacy monitoring. When `false`, no scheduled pulls will occur regardless of the frequency setting.
            </ResponseField>
            
            <ResponseField name="frequency" type="string" optional>
              How often scheduled data pulls should occur, can be one of `weekly` or `biweekly` or `monthly`.
            </ResponseField>
            
            <Info>
              See [When do scheduled pulls run?](/medical-api/handling-data/cohorts#when-do-scheduled-pulls-run%3F) for timing details.
            </Info>
          </Expandable>
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="laboratory" type="object" optional>
      Laboratory notifications - subscribes patients for ongoing notifications of new lab results.  
      <Expandable title="laboratory object properties">
        <ResponseField name="notifications" type="boolean" optional>
          Enable/disable laboratory monitoring notifications.
        </ResponseField>
        <ResponseField name="schedule" type="object" optional>
          Enable laboratory data pulls at scheduled dates.
          <Expandable title="schedule object properties" defaultOpen={false}>
            <ResponseField name="enabled" type="boolean" optional>
              Enable or disable scheduled data pulls for laboratory monitoring. When `false`, no scheduled pulls will occur regardless of the frequency setting.
            </ResponseField>
            
            <ResponseField name="frequency" type="string" optional>
              How often scheduled data pulls should occur, can be one of `weekly` or `biweekly` or `monthly`.
            </ResponseField>
            
            <Info>
              See [When do scheduled pulls run?](/medical-api/handling-data/cohorts#when-do-scheduled-pulls-run%3F) for timing details.
            </Info>
          </Expandable>
        </ResponseField>
      </Expandable>
    </ResponseField>
    <Info>
      For **pharmacy** and **laboratory** monitoring, choose either `notifications` OR `schedule`, but not both. If `notifications` is `true`, `schedule` value is ignored.
    </Info>
  </Expandable>
</ResponseField>
