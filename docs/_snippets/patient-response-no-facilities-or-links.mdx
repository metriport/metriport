</ResponseField>

</ResponseField>

<ResponseField name="lastName" type="string[]" required>
  The Patient's last name(s).
</ResponseField>

<ResponseField name="dob" type="string" required>
  The Patient's date of birth (DOB), formatted `YYYY-MM-DD` as per ISO 8601.
</ResponseField>

  The Patient's gender at birth, can be one of `M` or `F`.
</ResponseField>
